# Tooltip 文字提示（开发中）

## 基本用法

```tsx
import React from 'react';
import { Tooltip } from 'idesign-react';

const App = () => {
  return (
    <div className="idesign-demo-block-column">
      <div className="idesign-demo-block-row">
        <Tooltip>填充按钮</Tooltip>
      </div>
    </div>
  );
};

export default App;
```
