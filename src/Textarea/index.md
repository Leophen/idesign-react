# Textarea 多行文本框（开发中）

## 基本用法

import { Textarea } from 'idesign-react';

const App = () => {
  return (
    <div className="idesign-demo-block-column">
      <div className="idesign-demo-block-row">
        <Textarea>填充按钮</Textarea>
      </div>
    </div>
  );
};

export default App;
