## 例子

```jsx
import React from 'react';
import ReactDOM from 'react-dom';
import { Button } from 'idesign';

function App() {
  return (
    <Button type="primary">
      Hello World
    </Button>
  );
}

ReactDOM.render(<App />, document.getElementById('app'));
```
